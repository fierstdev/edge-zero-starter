```ts script
const anchorId = block.anchorId;
const variant = block.variant || 'banner';
const badge = block.badge;
const heading = block.heading;
const subheading = block.subheading;
const primaryAction = block.primaryAction;
const secondaryAction = block.secondaryAction;
const finePrint = block.finePrint;
const backgroundImage = block.backgroundImage;
const panelTitle = block.panelTitle;
const panelItems = Array.isArray(block.panelItems) ? block.panelItems : [];
const blockIndex = Number(block.blockIndex || 0);

const badgeClass = badge?.color ? `badge-${badge.color}` : 'badge-primary';
const sectionId = anchorId || `${block?.id || 'cta'}-section`;
const headingId = `${sectionId}-heading`;
const backgroundImageUrl = resolveMediaUrl(backgroundImage, ['feature', 'card', 'thumbnail', 'hero']);
const backgroundImageSrcSet = resolveMediaSrcSet(backgroundImage, ['thumbnail', 'card', 'feature', 'hero']);
const backgroundImageDimensions = resolveMediaDimensions(backgroundImage, ['feature', 'hero']);
const isPriorityMedia = blockIndex === 0;
const bannerSizes = '(min-width: 1200px) 1152px, 100vw';

const toButtonClass = (action?: any, isPrimary?: boolean) => {
  const defaultStyle = isPrimary ? 'btn-primary' : 'btn-outline';
  const styleClass = action?.style ? `btn-${action.style}` : defaultStyle;
  const sizeClass = action?.size ? `btn-${action.size}` : 'btn-md';
  return `btn ${styleClass} ${sizeClass} w-full sm:w-auto`;
};
```
