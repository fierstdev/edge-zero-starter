```ts script
const anchorId = block.anchorId;
const variant = block.variant || 'grid';
const badge = block.badge;
const heading = block.heading;
const subheading = block.subheading;
const columns = block.columns;
const spotlightMedia = block.spotlightMedia;
const spotlightMediaPosition = block.spotlightMediaPosition;
const features = block.features;

const badgeClass = badge?.color ? `badge-${badge.color}` : 'badge-primary';
const sectionId = anchorId || `${block?.id || 'features'}-section`;
const headingId = `${sectionId}-heading`;
const featureItems = Array.isArray(features) ? features : [];
const spotlightMediaUrl = resolveMediaUrl(spotlightMedia, ['feature', 'card', 'thumbnail', 'hero']);
const spotlightMediaSrcSet = resolveMediaSrcSet(spotlightMedia, ['thumbnail', 'card', 'feature', 'hero']);
const spotlightMediaDimensions = resolveMediaDimensions(spotlightMedia, ['card', 'feature', 'hero']);
const useLeftMedia = spotlightMediaPosition === 'left';
const [spotlightPrimary, ...spotlightSecondary] = featureItems;

const gridClass =
  columns === '2'
    ? 'md:grid-cols-2'
    : columns === '4'
      ? 'md:grid-cols-2 xl:grid-cols-4'
      : 'md:grid-cols-2 lg:grid-cols-3';

const toButtonClass = (action?: any) => {
  const styleClass = action?.style ? `btn-${action.style}` : 'btn-outline';
  const sizeClass = action?.size ? `btn-${action.size}` : 'btn-sm';
  return `btn ${styleClass} ${sizeClass}`;
};
```
